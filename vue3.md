import { useStore } from 'vuex'
import { computed } from 'vue'
export default {
  // vue3写法
  setup () {
    const store = useStore()
    return {
      num2: computed(() =>
        store.state.num2
      ),
      double2: computed(() => store.getters.double2),
      cnum2: (newnum) => {
        store.commit('changeNum2', newnum)
      },
      canum2: () => {
        store.dispatch('timecnum2')
      }
